<div>DPAD_UP = Open / Scroll Up</div>
<div>DPAD_DOWN = Scroll Down</div>
<div>DPAD_RIGHT = Select</div>
<div>DPAD_LEFT = Back / Exit</div>
<div>RB_BUTTON = Move Toggle Right</div>
<div>LB_BUTTON = Move Toggle Left</div>
